program Operators;
var x, y: integer;
begin
  x := 10 + 5 - 3 * 2;
  y := x div 2;
  x := y MOD 3;
  x := 10 / 2;
  if (x > 5) AND (y <= 10) then
    writeln('OK');
  if x == y then
    writeln('Equal');
  if x <> y then
    writeln('Not Equal');
  if (x >= 1) OR (y < 100) then
    writeln('Pass');
  if NOT (x == 0) then
    writeln('NonZero');
end.
