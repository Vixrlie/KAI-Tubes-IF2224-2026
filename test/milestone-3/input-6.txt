program TypesAndConsts;

const
  MaxSize == 50;
  Pi == 3.14;

type
  Index == 1..100;
  Color == (Red, Green, Blue);
  Matrix == array[1..3] of integer;

var
  idx: Index;
  c: Color;
  m: Matrix;

begin
  idx := 1;
  m[1] := MaxSize;
  m[2] := MaxSize * 2;
  m[3] := m[1] + m[2];

  if idx < 10 then
    if idx > 5 then
      writeln('mid')
    else
      writeln('low')
  else
    writeln('high');
end.
